### 
```sql
WITH CTE_status_sales as (
	SELECT
		*
	FROM production_status as ps
	LEFT JOIN sales_product as sp
	ON ps.product_id = sp.product_id
), CTE_sum_sales as (
		SELECT
			status,
		SUM(total_sales) OVER(PARTITION BY status)
		FROM CTE_status_sales
)
SELECT DISTINCT
 *
FROM CTE_sum_sales
```
